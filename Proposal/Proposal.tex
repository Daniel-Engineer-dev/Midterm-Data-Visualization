\section{Mô tả Dữ liệu}

\subsection{Nguồn dữ liệu}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Tên dataset:} E-commerce Sales Transactions Dataset
    \item \textbf{Nguồn:} Kaggle (tác giả: Arif Miah)
    \item \textbf{Link tải:} \url{https://www.kaggle.com/datasets/miadul/e-commerce-sales-transactions-dataset}
    \item \textbf{Mô tả:} Tập dữ liệu mô phỏng các giao dịch bán hàng thương mại điện tử, bao gồm \textbf{34.500 bản ghi} với \textbf{17 thuộc tính}. Dữ liệu được thiết kế để phản ánh thực tế mua sắm trực tuyến, hỗ trợ các tác vụ phân tích bán hàng, phân khúc khách hàng, dự đoán hoàn trả và đánh giá hiệu suất sản phẩm.
    \item \textbf{Giấy phép:} Apache 2.0
\end{itemize}

\subsection{Đặc điểm nổi bật}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Quy mô:} 34.500 bản ghi giao dịch
    \item \textbf{Số thuộc tính:} 17 biến (bao gồm cả thông tin đơn hàng, sản phẩm, khách hàng và vận chuyển)
    \item \textbf{Danh mục sản phẩm:} 7 danh mục (Electronics, Fashion, Home, Beauty, Sports, Toys, Grocery)
    \item \textbf{Phương thức thanh toán:} 6 loại (Credit Card, Debit Card, UPI, PayPal, COD, Wallet)
    \item \textbf{Thông tin nhân khẩu học:} Tuổi (18--70), giới tính (Male/Female/Other)
    \item \textbf{Bao gồm cả giao dịch hoàn trả} $\rightarrow$ cho phép phân tích tỷ lệ hoàn trả theo danh mục, khu vực
\end{itemize}

\subsection{Phân tích Why (Nhiệm vụ / Câu hỏi phân tích)}

Dưới đây là các câu hỏi phân tích dự kiến mà Dashboard sẽ trả lời:

\subsubsection{Nhóm 1: Xu hướng Doanh thu theo Thời gian}

\begin{enumerate}[leftmargin=1.5cm]
    \item \textbf{Doanh thu thay đổi như thế nào theo từng tháng/quý?} Có xu hướng tăng trưởng hay suy giảm không?
    \item \textbf{Tháng nào trong năm có doanh thu cao nhất?} Có yếu tố mùa vụ (seasonality) rõ ràng không?
    \item \textbf{Ngày nào trong tuần có nhiều đơn hàng nhất?} Xu hướng mua sắm theo thời gian ra sao?
\end{enumerate}

\subsubsection{Nhóm 2: Phân tích Sản phẩm \& Danh mục}

\begin{enumerate}[leftmargin=1.5cm, start=4]
    \item \textbf{Danh mục nào có doanh thu cao nhất?} So sánh doanh thu giữa 7 danh mục (Electronics, Fashion, Home, Beauty, Sports, Toys, Grocery).
    \item \textbf{Tỷ lệ hoàn trả theo từng danh mục là bao nhiêu?} Danh mục nào có tỷ lệ hoàn trả cao nhất?
    \item \textbf{Mức giảm giá ảnh hưởng thế nào đến doanh thu và biên lợi nhuận?}
\end{enumerate}

\subsubsection{Nhóm 3: Phân tích Khách hàng}

\begin{enumerate}[leftmargin=1.5cm, start=7]
    \item \textbf{Phân bố khách hàng theo độ tuổi và giới tính như thế nào?} Nhóm nào chi tiêu nhiều nhất?
    \item \textbf{Phương thức thanh toán nào được ưa chuộng nhất?} Có sự khác biệt theo nhóm tuổi/giới tính không?
    \item \textbf{Top khách hàng có giá trị cao nhất (theo tổng chi tiêu) là ai?}
\end{enumerate}

\subsubsection{Nhóm 4: Phân bố Khu vực \& Vận chuyển}

\begin{enumerate}[leftmargin=1.5cm, start=10]
    \item \textbf{Doanh thu phân bố ra sao theo các khu vực địa lý?} Khu vực nào có hiệu suất tốt nhất?
    \item \textbf{Thời gian giao hàng và chi phí vận chuyển khác nhau thế nào giữa các khu vực?}
    \item \textbf{Có mối tương quan nào giữa chi phí vận chuyển và tỷ lệ hoàn trả không?}
\end{enumerate}

\subsection{Tương tác dự kiến}

\begin{itemize}[leftmargin=1.5cm]
    \item \textbf{Filter / Slicer:} Lọc theo khu vực, khoảng thời gian, danh mục sản phẩm, phương thức thanh toán.
    \item \textbf{Hover Tooltip:} Hiển thị thông tin chi tiết khi di chuột vào biểu đồ.
    \item \textbf{Cross-filtering:} Click vào một khu vực hoặc danh mục $\rightarrow$ tự động cập nhật các biểu đồ khác.
    \item \textbf{Drill-down:} Từ tổng quan năm $\rightarrow$ quý $\rightarrow$ tháng $\rightarrow$ tuần.
\end{itemize}
